private void writeFile(String fileName, String info) {
　　  try {
　　   FileOutputStream fout = openFileOutput(fileName, MODE_PRIVATE);
　　   byte[] bytes = info.getBytes();
　　   fout.write(bytes);
　　   fout.close();
　　  } catch (Exception err) {
　　  }
　  }
